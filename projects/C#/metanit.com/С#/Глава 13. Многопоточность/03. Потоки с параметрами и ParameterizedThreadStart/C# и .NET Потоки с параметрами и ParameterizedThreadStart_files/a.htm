                                    Но что делать, если нам надо передать не один, а несколько параметров различного типа?
                                    Вместо классов можно предложить работу с кортежами:
                                    static void Main(string[] args)
                                    {
                                    (int x, int y) tuple = (5, 10);
                                    Thread myThread = new Thread(new ParameterizedThreadStart(Count));
                                    myThread.Start(tuple);
                                    }

                                    public static void Count(object x)
                                    {
                                    (int x, int y) tup = (System.ValueTuple<int, int="">)x;

                                    for (int i = 1; i < 9; i++)
                                    {
                                    Console.WriteLine("Второй поток:");
                                    Console.WriteLine( tup.x * tup.y * i);
                                    Thread.Sleep(400);
                                    }
                                    }
