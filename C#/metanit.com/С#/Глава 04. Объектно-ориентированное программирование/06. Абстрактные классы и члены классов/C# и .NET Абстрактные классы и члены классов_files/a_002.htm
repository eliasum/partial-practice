                                                                                Также очень удобно использовать абстрактые методы, реализуя обычные методы в абстрактном классе, которые могут скрывать свою реализацию от дочерних классов, но давать возможность использовать методы классов-потомков в данном методе. Например, это похоже на то, чтобы выполнить сложную работу или цепочку собственных методов или обработок, и базовый класс обращается к потомку за помощью в определённый момент.

                                                                                class Program
                                                                                {
                                                                                static void Main(string[] args)
                                                                                {
                                                                                var child = new Child();
                                                                                var child2 = new Child2();
                                                                                }
                                                                                }

                                                                                public abstract class Base
                                                                                {
                                                                                public abstract void Put();

                                                                                private void DoPrivateWork()
                                                                                {
                                                                                Do1();
                                                                                Put();
                                                                                Do2();
                                                                                }

                                                                                protected Base()
                                                                                {
                                                                                DoPrivateWork();
                                                                                }

                                                                                private void Do1()
                                                                                {
                                                                                Console.WriteLine("PUT FROM PRIVATE PARENT REALIZATION 1");
                                                                                }

                                                                                private void Do2()
                                                                                {
                                                                                Console.WriteLine("PUT FROM PRIVATE PARENT REALIZATION 2");
                                                                                }
                                                                                }

                                                                                public class Child : Base
                                                                                {
                                                                                public override void Put()
                                                                                {
                                                                                Console.WriteLine("PUT FROM CHILD REALIZATION");
                                                                                }
                                                                                }

                                                                                public class Child2 : Base
                                                                                {
                                                                                public override void Put()
                                                                                {
                                                                                Console.WriteLine("PUT FROM CHILD2 REALIZATION");
                                                                                }
                                                                                }

                                                                                Результат:
                                                                                PUT FROM PRIVATE PARENT REALIZATION 1
                                                                                PUT FROM CHILD REALIZATION
                                                                                PUT FROM PRIVATE PARENT REALIZATION 2

                                                                                PUT FROM PRIVATE PARENT REALIZATION 1
                                                                                PUT FROM CHILD2 REALIZATION
                                                                                PUT FROM PRIVATE PARENT REALIZATION 2

                                                                                                                                            нельзя создавать объекты абстрактных классов, то есть невозможно использовать конструктор для создания объекта абстрактного класса. Для создания переменной абстрактного класса необходимо использовать конструкторы производных классов, например, Account acc = new DepositAccount(). Таким образом, переменная Account всегда будет представлять объект одного из производных классов
